int solve()
{
    int n, m;
    cin >> n >> m;

    for(int i = 1; i <= n; i ++)
    {
        for(int j = 1; j <= m; j ++)
        {
            cin >> a[i][j];
            if(a[i][j] != 0) b[i][j] = true;
            else b[i][j] = false;
        }
    }
    queue<pair<int, int>> q;
    int ans = 0;
    for(int i = 1; i <= n; i ++)
    {
        for(int j = 1; j <= m; j ++)
        {
            if(!b[i][j]) continue;
            
            int now = 0;
            q.push(make_pair(i, j));
            b[i][j] = false;
            while(!q.empty())
            {
                pair<int, int> p;
                p = q.front();
                now += a[p.first][p.second];
                q.pop();
                for(int k = 0; k < 4; k ++)

                    }
                }
            }

#define pi acos(1.0)
#define e exp(-1.0)
using namespace std;
typedef long long LL;
const int mod = 1e9 + 7;

int solve()
{
    int n, k;
    cin >> n >> k;
    vector<int> a(n + 1);
    vector<vector<int>> dp(n + 1, vector<int>(1 << 6, 0));
    for(int i = 1; i <= n; i ++)
    {
        cin >> a[i];
        for(int mask = 0; mask < (1 << 6); mask ++) 
        {
            dp[i][mask] += dp[i - 1][mask];
            if(dp[i][mask] >= mod) dp[i][mask] -= mod;
            dp[i][mask & a[i]] += dp[i - 1][mask];
            if(dp[i][mask & a[i]] >= mod) dp[i][mask & a[i]] -= mod;
        }
        dp[i][a[i]] = (dp[i][a[i]] + 1) % mod;
    }
    int ans = 0;
    for(int mask = 0; mask < (1 << 6); mask ++)
    {
        if(__builtin_popcount(mask) == k)
        {
            ans = (ans + dp[n][mask]) % mod;
        }
    }
    return ans;
}
int main()
{
    int t;
    cin >> t;
    while(t --)
    {
        cout << solve() << endl;
